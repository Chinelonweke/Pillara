# requirements.txt
#
# WHY PIN EXACT VERSIONS:
# "fastapi" without a version means pip installs the latest.
# If FastAPI releases a breaking change tomorrow, your deployment breaks.
# Pinning versions means your app behaves identically on every machine —
# your laptop, your teammate's laptop, your EC2 server, your Docker container.
#
# HOW TO UPDATE SAFELY:
# 1. Change the version number here
# 2. Run: pip install -r requirements.txt
# 3. Run your tests
# 4. If tests pass, the update is safe to deploy
#
# GROUPS EXPLAINED:
# Each group is labelled by what it does in Pillara.
# If you remove a feature, you know exactly which packages to remove.

# ─── SECURITY ─────────────────────────────────────────────────────────────────
passlib[bcrypt]==1.7.4          # password hashing — bcrypt algorithm
                                # [bcrypt] installs the bcrypt C extension
bcrypt==4.0.1                   # PINNED: passlib 1.7.4 reads bcrypt.__about__.__version__
                                # to detect the backend. bcrypt 4.1.1+ removed that
                                # attribute, which makes passlib's version detection
                                # silently fail and fall back to a buggy code path that
                                # doesn't truncate passwords to bcrypt's 72-byte limit —
                                # causing "password cannot be longer than 72 bytes" even
                                # for short passwords. passlib is largely unmaintained,
                                # so the practical fix is pinning bcrypt to the last
                                # version known to still expose __about__.
